# Determine remaining deductible & max out of pocket
def determine_deductible(
    deductible, max_pocket, ded_diff, coinsurance, amount, options
):
    """Take an amount and determine if it meets the deductible.
    If it doesn't meet the deductible, subtract the difference.
    If the deductible is met, determine if the max out of pocket is met otherwise
    subtract the difference

    :param deductible: Remaining deductible.
    :type deductible: float
    :param max_pocket: Remaining max out of pocket.
    :type max_pocket: float
    :param ded_diff: Difference between deductible and max out of pocket.
    :type ded_diff: float
    :param coinsurance: % for coinusrance.
    :type coinusrance: float
    :param amount: Medical expenses to deduct.
    :type amout: float
    :param options: Additional options.
    :type options: NA

    :return: deductible, max_pocket
    :rtype: float, float
    """

    if max_pocket != 0:
        # Subtract from deductible
        if amount < deductible:
            deductible = deductible - amount
        if amount >= deductible:
            amount = amount - deductible
            amount = amount * (1 - coinsurance)

            print("Expense after coinsurance " + str(amount))

            deductible = 0

            if amount < ded_diff:
                max_pocket = max_pocket - amount
            if amount >= ded_diff:
                max_pocket = 0

    return deductible, max_pocket

# Create function to calculate cost
def insurance_cost(plan, office_visit, options):
    """Will determine how much health insurance costs.

    :param plan: Insurance plan info.
    :type plan: dict {
        deductible: Total $ deductible for the year
        coinsure: Coinusrance % (float)
        out_pocket: Total $ out of pocket maximum (float)
        office: Total $ primary office visit that is not annual exam (float)
        specialist: Total $ for specialist office visit (float)
        lab: % after deductible is met (float)
        urgent: Total Urgent care cost (float)
        drugs1: Total $ or % of deductible (float)
        drugs2: Total $ or % of deductible (float)
        drugs3: Total $ or % of deductible (float)
        rates: Weekly premium (float)
    }
    :param office_visit: Office visit/expenses for the year.
    :type office_visit: dict {
        primary: Number of times for a primary office visit (int)
        specialist: Number of times for a specialist visit (int)
        primary_hsa: HSA cost for primary office visit (float)
        specialist_hsa: HSA cost for specialist visit (float)
        urgent: Number of times for a urgent care visit (int)
        urgent_hsa: HSA cost for specialist visit (float)
        lab: Lab costs (float)
    }
    :param options: Additional options.
    :type options: dict {
        additional: Additional cost that can be applied to deducitible (float)
        hsa_account: How much money in HSA account (float)
    }

    :return: total_cost
    :rtype: float
    """
    deductible = plan["deductible"]
    coinsurance = plan["coinsure"]
    out_pocket = plan["out_pocket"]  # - deductible
    lab_costs = office_visit["lab"]
    additional_ded = options["additional"]
    hsa_account = options["hsa_account"]

    total_ded = out_pocket + deductible

    total_cost = 0

    # Difference between deductible  & out_pocket
    ded_diff = out_pocket - deductible

    # Get premium cost
    total_premium = plan["rates"] * 52

    # Add office visit costs
    total_office = 0
    if plan["office"] > 1:
        total_office = (
            total_office
            + plan["office"] * office_visit["primary"]
            + plan["specialist"] * office_visit["specialist"]
            + plan["urgent"] * office_visit["urgent"]
        )
        total_cost = total_cost + total_office

    else:
        total_office = (
            total_office
            + office_visit["primary_hsa"] * office_visit["primary"]
            + office_visit["specialist_hsa"] * office_visit["specialist"]
            + office_visit["urgent_hsa"] * office_visit["urgent"]
        )

        options = "NA"
        deductible, out_pocket = determine_deductible(
            deductible, out_pocket, ded_diff, coinsurance, total_office, options
        )
    # Add lab costs
    deductible, out_pocket = determine_deductible(
        deductible, out_pocket, ded_diff, coinsurance, lab_costs, options
    )

    # Add additional costs
    if additional_ded > 0:
        deductible, out_pocket = determine_deductible(
            deductible, out_pocket, ded_diff, coinsurance, additional_ded, options
        )

    # Determine total cost by subtracting dedutible and out of pocket max
    total_cost = total_cost + deductible + out_pocket - total_ded

    # If HSA account subtract from total
    total_cost = total_cost - hsa_account

    return total_cost
